   ```rust
   #[command(name = "memd", version, about = "Agent memory service")]
   struct Args {
       #[arg(long, default_value = "mcp")]
       mode: String,  // "mcp" or "cli"

       #[arg(long)]
       config: Option<PathBuf>,

       #[command(subcommand)]
       command: Option<CliCommand>,
   }

   #[tokio::main]
   async fn main() -> Result<()> {
       let args = Args::parse();
       let config = load_config(args.config.as_deref())?;
       init_logging(&config.log_format, &config.log_level);

       let store = Arc::new(MemoryStore::new());

       match args.mode.as_str() {
           "mcp" => {
               let mut server = McpServer::new(config, store);
               server.run().await
           }
           "cli" => {
               if let Some(cmd) = args.command {
                   run_cli(&*store, cmd).await
               } else {
                   eprintln!("CLI mode requires a subcommand. Use --help for usage.");
                   Ok(())
               }
           }
           _ => {
               eprintln!("Unknown mode: {}. Use 'mcp' or 'cli'.", args.mode);
               Ok(())
           }
       }
   }
   ```

3. `evals/harness/src/mcp_client.rs`:
   ```rust
   use std::process::{Child, Command, Stdio};
   use std::io::{BufRead, BufReader, Write};

   pub struct McpClient {
       process: Child,
       stdin: std::process::ChildStdin,
       stdout: BufReader<std::process::ChildStdout>,
       request_id: i64,
   }

   impl McpClient {
       /// Start memd as a subprocess
       pub fn start(memd_path: &str) -> Result<Self> {
           let mut process = Command::new(memd_path)
               .arg("--mode")
               .arg("mcp")
               .stdin(Stdio::piped())
               .stdout(Stdio::piped())
               .stderr(Stdio::null())  // Suppress logs in tests
               .spawn()?;

           let stdin = process.stdin.take().unwrap();
           let stdout = BufReader::new(process.stdout.take().unwrap());

           Ok(Self { process, stdin, stdout, request_id: 0 })
       }

       /// Send request and get response
       pub fn request(&mut self, method: &str, params: Option<Value>) -> Result<Value> {
           self.request_id += 1;
           let request = json!({
               "jsonrpc": "2.0",
               "id": self.request_id,
               "method": method,
               "params": params
           });

           writeln!(self.stdin, "{}", request)?;
           self.stdin.flush()?;

           let mut line = String::new();
           self.stdout.read_line(&mut line)?;
           let response: Value = serde_json::from_str(&line)?;

           Ok(response)
       }

       /// Initialize connection
       pub fn initialize(&mut self) -> Result<Value> {
           self.request("initialize", None)
       }

       /// List tools
       pub fn tools_list(&mut self) -> Result<Value> {
           self.request("tools/list", None)
       }

       /// Call a tool
       pub fn call_tool(&mut self, name: &str, arguments: Value) -> Result<Value> {
           self.request("tools/call", Some(json!({
               "name": name,
               "arguments": arguments
           })))
       }
   }

   impl Drop for McpClient {
       fn drop(&mut self) {
           let _ = self.process.kill();
       }
   }
   ```

   ```rust
   use clap::Parser;

   mod mcp_client;
   mod suites;

   #[derive(Parser)]
   struct Args {
       /// Path to memd binary
       #[arg(long, default_value = "target/debug/memd")]
       memd_path: String,

       /// Suite to run (all, mcp, isolation, recovery)
       #[arg(long, default_value = "all")]
       suite: String,

       /// Output format (json, pretty)
       #[arg(long, default_value = "pretty")]
       output: String,
   }

   #[tokio::main]
   async fn main() {
       let args = Args::parse();

       // Build memd first
       println!("Building memd...");
       let status = std::process::Command::new("cargo")
           .args(["build", "-p", "memd"])
           .status()
           .expect("Failed to build memd");

       if !status.success() {
           eprintln!("Build failed");
           std::process::exit(1);
       }

       let results = match args.suite.as_str() {
           "all" => {
               let mut all = vec![];
               all.extend(suites::mcp_conformance::run(&args.memd_path).await);
               all
           }
           "mcp" => suites::mcp_conformance::run(&args.memd_path).await,
           _ => {
               eprintln!("Unknown suite: {}", args.suite);
               std::process::exit(1);
           }
       };

       // Print results
       let passed = results.iter().filter(|r| r.passed).count();
       let total = results.len();

       println!("\n{}/{} tests passed", passed, total);

       for result in &results {
           let status = if result.passed { "PASS" } else { "FAIL" };
           println!("  [{}] {}", status, result.name);
           if !result.passed {
               println!("       {}", result.message);
           }
       }

       if passed != total {
           std::process::exit(1);
       }
   }
   ```

   ```rust
   fn test_initialize(memd_path: &str) -> TestResult {
       // EVAL-02: Initialize returns protocol version and capabilities
       let mut client = match McpClient::start(memd_path) {
           Ok(c) => c,
           Err(e) => return TestResult::fail("A1_initialize", &format!("Failed to start: {}", e)),
       };

       match client.initialize() {
           Ok(response) => {
               let has_version = response.get("result")
                   .and_then(|r| r.get("protocolVersion"))
                   .is_some();
               let has_capabilities = response.get("result")
                   .and_then(|r| r.get("capabilities"))
                   .is_some();

               if has_version && has_capabilities {
                   TestResult::pass("A1_initialize")
               } else {
                   TestResult::fail("A1_initialize", "Missing protocolVersion or capabilities")
               }
   ```
