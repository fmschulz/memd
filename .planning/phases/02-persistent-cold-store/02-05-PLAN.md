//! Segment reader with memory-mapped I/O
//!
//! Reads finalized segments efficiently using mmap.
//! Tombstone filtering ensures deleted chunks return None.

use std::fs::File;
use std::path::{Path, PathBuf};

use memmap2::{Mmap, MmapOptions};

use super::format::{PayloadIndexRecord, SegmentMeta};
use crate::error::{MemdError, Result};
use crate::store::TombstoneSet;

/// Reader for a finalized segment
pub struct SegmentReader {
    pub id: u64,
    dir: PathBuf,
    payload_mmap: Mmap,
    index: Vec<PayloadIndexRecord>,
    tombstones: TombstoneSet,
}

impl SegmentReader {
    /// Open a finalized segment for reading
    pub fn open(dir: PathBuf) -> Result<Self> {
        // Extract segment ID from directory name (seg_NNNNNN)
        let id = Self::parse_segment_id(&dir)?;

        // Open payload file and mmap
        let payload_path = dir.join("payload.bin");
        let payload_file = File::open(&payload_path)
            .map_err(|e| MemdError::Storage(format!("open payload.bin: {}", e)))?;

        // SAFETY: File is finalized and immutable (append-only, no truncation)
        let payload_mmap = unsafe {
            MmapOptions::new()
                .map(&payload_file)
                .map_err(|e| MemdError::Storage(format!("mmap payload: {}", e)))?
        };

        // Load index
        let index_path = dir.join("payload.idx");
        let index = Self::load_index(&index_path)?;

        // Load tombstones
        let tombstone_path = dir.join("tombstone.bin");
        let tombstones = TombstoneSet::load_or_create(tombstone_path)?;

        Ok(Self {
            id,
            dir,
            payload_mmap,
            index,
            tombstones,
        })
    }

    fn parse_segment_id(dir: &Path) -> Result<u64> {
        let name = dir
            .file_name()
            .and_then(|n| n.to_str())
            .ok_or_else(|| MemdError::Storage("invalid segment dir".into()))?;

        // Format: seg_NNNNNN
        if !name.starts_with("seg_") {
            return Err(MemdError::Storage(format!("invalid segment name: {}", name)));
        }

        name[4..]
            .parse()
            .map_err(|_| MemdError::Storage(format!("invalid segment id: {}", name)))
    }

    fn load_index(path: &Path) -> Result<Vec<PayloadIndexRecord>> {
        let bytes = std::fs::read(path)
            .map_err(|e| MemdError::Storage(format!("read index: {}", e)))?;

        PayloadIndexRecord::parse_all(&bytes)
    }

    /// Number of chunks in this segment (including tombstoned)
    pub fn chunk_count(&self) -> u32 {
        self.index.len() as u32
    }

    /// Number of active (non-tombstoned) chunks
    pub fn active_count(&self) -> u32 {
        self.chunk_count() - self.tombstones.deleted_count() as u32
    }

    /// Read chunk by ordinal, returns None if tombstoned or invalid
    ///
    /// Verifies CRC-32 on read to detect corruption.
    pub fn read_chunk(&self, ordinal: u32) -> Result<Option<Vec<u8>>> {
        // Check bounds
        if ordinal as usize >= self.index.len() {
            return Ok(None);
        }

        // Check tombstone
        if self.tombstones.is_deleted(ordinal) {
            return Ok(None);
        }

        let record = &self.index[ordinal as usize];
        let start = record.offset as usize;
        let end = start + record.length as usize;

        // Bounds check on mmap
        if end > self.payload_mmap.len() {
            return Err(MemdError::Storage(format!(
                "segment {} ordinal {} offset out of bounds",
                self.id, ordinal
            )));
        }

        let data = &self.payload_mmap[start..end];

        // Verify CRC-32
        let computed_crc = crc32fast::hash(data);
        if computed_crc != record.crc32 {
            return Err(MemdError::Storage(format!(
                "segment {} ordinal {} CRC mismatch: expected {:08x}, got {:08x}",
                self.id, ordinal, record.crc32, computed_crc
            )));
        }

        Ok(Some(data.to_vec()))
    }

    /// Mark ordinal as deleted (updates tombstone)
    pub fn mark_deleted(&mut self, ordinal: u32) -> Result<()> {
        self.tombstones.mark_deleted(ordinal);
        self.tombstones.persist()
    }

    /// Get segment directory path
    pub fn dir(&self) -> &Path {
        &self.dir
    }

    /// Load segment metadata
    pub fn metadata(&self) -> Result<SegmentMeta> {
        let meta_path = self.dir.join("meta");
        SegmentMeta::load(&meta_path)
    }
}

        Ok(records)
    }

//! WAL reader for crash recovery
//!
//! Reads WAL records and supports replaying for recovery.

use std::fs::File;
use std::io::Read;
use std::path::Path;

use super::format::{WalRecord, WalRecordType, WAL_MAGIC, HEADER_SIZE};
use crate::error::{MemdError, Result};

/// WAL reader
pub struct WalReader {
    data: Vec<u8>,
}

impl WalReader {
    /// Open and read entire WAL file
    pub fn open(path: &Path) -> Result<Self> {
        if !path.exists() {
            return Ok(Self { data: Vec::new() });
        }

        let mut file = File::open(path)
            .map_err(|e| MemdError::Storage(format!("open WAL: {}", e)))?;
        let mut data = Vec::new();
        file.read_to_end(&mut data)
            .map_err(|e| MemdError::Storage(format!("read WAL: {}", e)))?;

        Ok(Self { data })
    }

    /// Check if WAL is empty
    pub fn is_empty(&self) -> bool {
        self.data.is_empty()
    }

    /// Read all valid records from WAL
    ///
    /// Stops at first corrupted record (partial write on crash).
    /// Returns records and count of bytes successfully read.
    pub fn read_all_records(&self) -> Result<(Vec<WalRecord>, usize)> {
        let mut records = Vec::new();
        let mut offset = 0;

        while offset + HEADER_SIZE <= self.data.len() {
            match WalRecord::decode_from_bytes(&self.data[offset..]) {
                Ok((record, consumed)) => {
                    records.push(record);
                    offset += consumed;
                }
                Err(_) => {
                    // Corrupted record - stop here (partial write on crash)
                    break;
                }
            }
        }

        Ok((records, offset))
    }

    /// Find the last checkpoint record
    pub fn find_last_checkpoint(&self) -> Result<Option<usize>> {
        let (records, _) = self.read_all_records()?;

        Ok(records
            .iter()
            .rposition(|r| matches!(r.record_type, WalRecordType::Checkpoint)))
    }

    /// Get records for recovery (after last checkpoint)
    ///
    /// Returns records that need to be replayed.
    pub fn records_for_recovery(&self) -> Result<Vec<WalRecord>> {
        let (records, _) = self.read_all_records()?;

        let checkpoint_idx = records
            .iter()
            .rposition(|r| matches!(r.record_type, WalRecordType::Checkpoint))
            .map(|i| i + 1)  // Start after checkpoint
            .unwrap_or(0);   // No checkpoint = replay all

        Ok(records[checkpoint_idx..].to_vec())
    }
}

/// Recovery helper functions
pub mod recovery {
    use super::*;

    /// Callback type for applying recovered records
    pub trait RecoveryHandler {
        /// Called for each Add record
        fn on_add(&mut self, record: &WalRecord) -> Result<()>;

        /// Called for each Delete record
        fn on_delete(&mut self, record: &WalRecord) -> Result<()>;

        /// Check if a chunk already exists (for idempotency)
        fn chunk_exists(&self, chunk_id: &str) -> Result<bool>;
    }

    /// Replay WAL records through handler
    ///
    /// Skips Add records for chunks that already exist (idempotent).
    pub fn replay<H: RecoveryHandler>(
        reader: &WalReader,
        handler: &mut H,
    ) -> Result<RecoveryStats> {
        let records = reader.records_for_recovery()?;

        let mut stats = RecoveryStats::default();

        for record in &records {
            match record.record_type {
                WalRecordType::Add => {
                    // Idempotent: skip if chunk already exists
                    if handler.chunk_exists(&record.chunk_id)? {
                        stats.skipped += 1;
                        continue;
                    }
                    handler.on_add(record)?;
                    stats.adds += 1;
                }
                WalRecordType::Delete => {
                    handler.on_delete(record)?;
                    stats.deletes += 1;
                }
                WalRecordType::Checkpoint => {
                    // Should not appear after last checkpoint
                    stats.checkpoints += 1;
                }
            }
        }

        Ok(stats)
    }

    #[derive(Debug, Default)]
    pub struct RecoveryStats {
        pub adds: usize,
        pub deletes: usize,
        pub skipped: usize,
        pub checkpoints: usize,
    }
}
