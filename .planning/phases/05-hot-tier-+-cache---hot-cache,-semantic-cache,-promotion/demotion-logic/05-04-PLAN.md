impl QueryMetrics {
    pub fn from_tiered_timing(
        timing: &TieredSearchTiming,
        embed_time: Duration,
        fetch_time: Duration,
        total_time: Duration,
    ) -> Self {
        let source = if timing.cache_hit {
            "cache"
        } else if timing.hot_sufficient {
            "hot"
        } else {
            "warm"
        };

        Self {
            embed_ms: embed_time.as_millis() as u64,
            dense_search_ms: (timing.hot_search_time + timing.warm_search_time).as_millis() as u64,
            fetch_ms: fetch_time.as_millis() as u64,
            total_ms: total_time.as_millis() as u64,
            cache_hit: timing.cache_hit,
            hot_search_ms: timing.hot_search_time.as_millis() as u64,
            warm_search_ms: timing.warm_search_time.as_millis() as u64,
            search_source: source.to_string(),

impl MetricsCollector {
    // Add tiered metrics field
    tiered: RwLock<TieredMetrics>,

    pub fn record_cache_hit(&self) {
        self.tiered.write().cache_hits += 1;
    }

    pub fn record_cache_miss(&self) {
        self.tiered.write().cache_misses += 1;
    }

    pub fn record_hot_search(&self, sufficient: bool) {
        let mut tiered = self.tiered.write();
        tiered.hot_searches += 1;
        if sufficient {
            tiered.hot_sufficient += 1;
        } else {
            tiered.warm_fallbacks += 1;
        }
    }

    pub fn record_promotion(&self) {
        self.tiered.write().promotions += 1;
    }

    pub fn record_eviction(&self) {
        self.tiered.write().evictions += 1;
    }

    pub fn get_tiered_metrics(&self) -> TieredMetrics {
        self.tiered.read().clone()
    }
}

pub async fn search_with_timing(
    &self,
    tenant_id: &TenantId,
    query: &str,
    k: usize,
    context: Option<SearchContext>,
) -> Result<(Vec<HybridSearchResult>, HybridTiming)> {
    let total_start = Instant::now();
    let mut timing = HybridTiming::default();

    // Get query embedding for tiered search
    let embed_start = Instant::now();
    let query_embedding = self.tiered.warm_searcher().embedder.embed_query(query).await?;
    let embed_time = embed_start.elapsed();

    // Step 1: Tiered dense search (cache -> hot -> warm)
    let dense_start = Instant::now();
    let active_project = context.as_ref().and_then(|c| c.current_project.as_deref());
    let (tiered_results, tiered_timing) = self.tiered
        .search_with_timing(tenant_id, &query_embedding, self.config.dense_k, active_project)
        .await?;
    timing.dense_time = dense_start.elapsed();

    // Convert tiered results to fusion candidates
    let dense_results: Vec<_> = tiered_results.into_iter()
        .enumerate()
        .map(|(rank, r)| FusionCandidate {
            chunk_id: r.chunk_id,
            source: FusionSource::Dense,
            rank: rank + 1,
            source_score: r.score,
        })
        .collect();

    // Log tiered search outcome for debug visibility
    tracing::debug!(
        cache_hit = tiered_timing.cache_hit,
        hot_sufficient = tiered_timing.hot_sufficient,
        hot_ms = tiered_timing.hot_search_time.as_millis(),
        warm_ms = tiered_timing.warm_search_time.as_millis(),
        "tiered dense search completed"
    );

    // Step 2: Sparse search (unchanged)
    // ... rest of method unchanged ...
}

async fn search_with_scores_real(
    &self,
    tenant_id: &TenantId,
    query: &str,
    k: usize,
) -> Result<Vec<(MemoryChunk, f32)>> {
    let total_start = Instant::now();

    if let Some(ref hybrid) = self.hybrid_searcher {
        let (hybrid_results, timing) = hybrid
            .search_with_timing(tenant_id, query, k, None)
            .await?;

        // Record tiered metrics
        if timing.cache_hit {
            self.metrics.record_cache_hit();
        } else {
            self.metrics.record_cache_miss();
            self.metrics.record_hot_search(timing.hot_sufficient);
        }

        // Fetch chunks and build results
        let fetch_start = Instant::now();
        let mut results = Vec::with_capacity(hybrid_results.len());
        for result in hybrid_results {
            if let Some(chunk) = self.get(tenant_id, &result.chunk_id).await? {
                results.push((chunk, result.final_score));
            }
        }
        let fetch_time = fetch_start.elapsed();

        // Record query metrics with tiered info
        self.metrics.record_query(QueryMetrics::from_tiered_timing(
            &timing.tiered_timing,  // From HybridTiming
            timing.embed_time,
            fetch_time,
            total_start.elapsed(),
        ));

        return Ok(results);
    }

}
