1. SemanticCacheConfig:
```rust
pub struct SemanticCacheConfig {
    pub similarity_threshold: f32,  // Default 0.88 (from CONTEXT.md: 0.85-0.95)
    pub ttl_secs: u64,              // Default 120 (from CONTEXT.md: 1-2 min)
    pub max_entries: u64,           // Default 10_000
}
```

2. CachedQueryResult:
```rust
#[derive(Clone)]
pub struct CachedQueryResult {
    pub chunk_ids: Vec<ChunkId>,
    pub scores: Vec<f32>,
    pub tenant_version: u64,  // For invalidation
    pub cached_at_ms: i64,
}
```

3. CacheEntry (internal):
```rust
struct CacheEntry {
    embedding: Vec<f32>,
    result: CachedQueryResult,
}
```

4. SemanticCache:
- cache: moka::sync::Cache<String, CacheEntry> keyed by tenant_id
- entries: parking_lot::RwLock<HashMap<String, Vec<CacheEntry>>> for similarity lookup
- config: SemanticCacheConfig

Key methods:

- new(config) -> Create cache with TTL from config
- get_similar(tenant_id, query_embedding, current_version) -> Option<CachedQueryResult>
  - Get all entries for tenant
  - Compute cosine similarity with each cached embedding
  - Return if similarity >= threshold AND version matches
  - Track cache hit in debug log
- insert(tenant_id, query_embedding, result) -> ()
  - Add entry to tenant's list
  - Entry auto-expires via moka TTL
- invalidate_tenant(tenant_id) -> ()
  - Remove all entries for tenant (called on chunk add/delete)
- stats() -> (hit_count, miss_count, entry_count)

Cosine similarity helper (vectors assumed normalized by embedder):
```rust
fn cosine_similarity(a: &[f32], b: &[f32]) -> f32 {
    debug_assert_eq!(a.len(), b.len());
    a.iter().zip(b.iter()).map(|(x, y)| x * y).sum::<f32>().clamp(-1.0, 1.0)
}
```

Implementation notes:
- Use moka::sync::Cache with time_to_live for automatic TTL
- Store entries in HashMap by tenant for efficient tenant-scoped lookup
- Log cache hits/misses with tracing::debug for observability
- Version check: reject if current_version > cached.tenant_version

#[cfg(test)]
mod tests {
    use super::*;

    fn make_embedding(seed: f32) -> Vec<f32> {
        // Simple normalized embedding for testing
        let mut v = vec![seed, 1.0 - seed, 0.5, 0.5];
        let norm: f32 = v.iter().map(|x| x * x).sum::<f32>().sqrt();
        for x in v.iter_mut() {
            *x /= norm;
        }
        v
    }

    #[test]
    fn test_cache_hit_on_similar_query() {
        let config = SemanticCacheConfig::default();
        let cache = SemanticCache::new(config);
        let tenant = "test_tenant".to_string();

        let embedding = make_embedding(0.8);
        let result = CachedQueryResult {
            chunk_ids: vec![ChunkId::new()],
            scores: vec![0.95],
            tenant_version: 1,
            cached_at_ms: 0,
        };

        cache.insert(&tenant, embedding.clone(), result.clone());

        // Exact same embedding should hit
        let hit = cache.get_similar(&tenant, &embedding, 1);
        assert!(hit.is_some());

        // Slightly different embedding should also hit (above threshold)
        let similar = make_embedding(0.81);  // Very similar
        let hit = cache.get_similar(&tenant, &similar, 1);
        assert!(hit.is_some());
    }

    #[test]
    fn test_cache_miss_on_different_query() {
        let config = SemanticCacheConfig::default();
        let cache = SemanticCache::new(config);
        let tenant = "test_tenant".to_string();

        let embedding1 = make_embedding(0.1);
        let embedding2 = make_embedding(0.9);  // Very different

        let result = CachedQueryResult {
            chunk_ids: vec![ChunkId::new()],
            scores: vec![0.95],
            tenant_version: 1,
            cached_at_ms: 0,
        };

        cache.insert(&tenant, embedding1, result);

        // Different embedding should miss
        let hit = cache.get_similar(&tenant, &embedding2, 1);
        assert!(hit.is_none());
    }

    #[test]
    fn test_version_invalidation() {
        let config = SemanticCacheConfig::default();
        let cache = SemanticCache::new(config);
        let tenant = "test_tenant".to_string();

        let embedding = make_embedding(0.5);
        let result = CachedQueryResult {
            chunk_ids: vec![ChunkId::new()],
            scores: vec![0.95],
            tenant_version: 1,
            cached_at_ms: 0,
        };

        cache.insert(&tenant, embedding.clone(), result);

        // Current version > cached version should invalidate
        let hit = cache.get_similar(&tenant, &embedding, 2);
        assert!(hit.is_none());
    }

    #[test]
    fn test_tenant_isolation() {
        let config = SemanticCacheConfig::default();
        let cache = SemanticCache::new(config);

        let embedding = make_embedding(0.5);
        let result = CachedQueryResult {
            chunk_ids: vec![ChunkId::new()],
            scores: vec![0.95],
            tenant_version: 1,
            cached_at_ms: 0,
        };

        cache.insert(&"tenant_a".to_string(), embedding.clone(), result);

        // Different tenant should not hit
        let hit = cache.get_similar(&"tenant_b".to_string(), &embedding, 1);
        assert!(hit.is_none());
    }
}
