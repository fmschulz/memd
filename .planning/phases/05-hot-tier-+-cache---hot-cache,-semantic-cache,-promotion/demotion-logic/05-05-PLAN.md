```rust
//! Tiered search evaluation suite
//!
//! Tests hot tier and semantic cache performance:
//! - Hot tier queries faster than warm tier
//! - Repeated queries hit semantic cache
//! - Promotion/demotion works correctly
//!
//! Quantitative thresholds:
//! - Cache hit rate >= 50% on repeated identical queries
//! - Latency improvement >= 20% from first to subsequent queries
//! - Tiered metrics present in memory.metrics response

use crate::mcp_client::McpClient;
use crate::{EvalResult, EvalSuite};
use std::time::Instant;

/// Minimum cache hit rate on repeated identical queries
const MIN_CACHE_HIT_RATE: f64 = 0.50;  // 50%

/// Minimum latency improvement from first to subsequent queries
const MIN_LATENCY_IMPROVEMENT: f64 = 0.20;  // 20%

pub struct TieredSuite;

impl EvalSuite for TieredSuite {
    fn name(&self) -> &'static str {
        "tiered"
    }

    fn run(&self, client: &mut McpClient) -> Vec<EvalResult> {
        let mut results = Vec::new();

        // Test 1: Cache hit on repeated query (>= 50% hit rate)
        results.push(test_semantic_cache_hit(client));

        // Test 2: Hot tier faster than cold start (>= 20% improvement)
        results.push(test_hot_tier_latency(client));

        // Test 3: Metrics report tiered stats
        results.push(test_tiered_metrics(client));

        results
    }
}

fn test_semantic_cache_hit(client: &mut McpClient) -> EvalResult {
    let tenant_id = "tiered_test_tenant";
    let test_name = "semantic_cache_hit";

    // Add a chunk
    let chunk_text = "The getUserById function retrieves user data from the database.";
    let add_result = client.call_tool("memory.add", serde_json::json!({
        "tenant_id": tenant_id,
        "text": chunk_text,
        "type": "code"
    }));

    if add_result.is_err() {
        return EvalResult::fail(test_name, "failed to add chunk");
    }

    // Warm up: first search (cold)
    let query = "getUserById function";
    let _ = client.call_tool("memory.search", serde_json::json!({
        "tenant_id": tenant_id,
        "query": query,
        "k": 5
    }));

    // Run multiple identical queries to test cache
    let num_queries = 5;
    for _ in 0..num_queries {
        let _ = client.call_tool("memory.search", serde_json::json!({
            "tenant_id": tenant_id,
            "query": query,
            "k": 5
        }));
    }

    // Check metrics for cache hit rate
    let metrics = client.call_tool("memory.metrics", serde_json::json!({
        "tenant_id": tenant_id
    }));

    if let Ok(m) = metrics {
        let tiered = m.get("tiered");
        if let Some(t) = tiered {
            let hits = t.get("cache_hits").and_then(|v| v.as_u64()).unwrap_or(0);
            let misses = t.get("cache_misses").and_then(|v| v.as_u64()).unwrap_or(0);
            let total = hits + misses;

            if total > 0 {
                let hit_rate = hits as f64 / total as f64;
                if hit_rate >= MIN_CACHE_HIT_RATE {
                    return EvalResult::pass(test_name, format!(
                        "cache hit rate {:.1}% >= {:.1}% threshold ({} hits / {} total)",
                        hit_rate * 100.0, MIN_CACHE_HIT_RATE * 100.0, hits, total
                    ));
                } else {
                    return EvalResult::fail(test_name, format!(
                        "cache hit rate {:.1}% < {:.1}% threshold ({} hits / {} total)",
                        hit_rate * 100.0, MIN_CACHE_HIT_RATE * 100.0, hits, total
                    ));
                }
            }
        }
    }

    // Cache hit not detected - infrastructure may not be wired
    EvalResult::warn(test_name, "cache metrics not available (tiered search may not be enabled)")
}

fn test_hot_tier_latency(client: &mut McpClient) -> EvalResult {
    let tenant_id = "tiered_latency_test";
    let test_name = "hot_tier_latency";

    // Add multiple chunks
    for i in 0..10 {
        let _ = client.call_tool("memory.add", serde_json::json!({
            "tenant_id": tenant_id,
            "text": format!("Function example_{} does something specific.", i),
            "type": "code"
        }));
    }

    // Search multiple times to build access patterns (for promotion)
    let query = "Function example";
    let mut latencies: Vec<u128> = Vec::new();

    for _ in 0..5 {
        let start = Instant::now();
        let _ = client.call_tool("memory.search", serde_json::json!({
            "tenant_id": tenant_id,
            "query": query,
            "k": 10
        }));
        latencies.push(start.elapsed().as_millis());
    }

    // Calculate improvement from first to average of last 3
    let first = latencies[0] as f64;
    let last_avg = latencies[2..].iter().map(|&x| x as f64).sum::<f64>() / 3.0;

    if first > 0.0 {
        let improvement = (first - last_avg) / first;

        if improvement >= MIN_LATENCY_IMPROVEMENT {
            EvalResult::pass(test_name, format!(
                "latency improved {:.1}% ({}ms -> {:.0}ms avg), >= {:.0}% threshold",
                improvement * 100.0, first as u128, last_avg, MIN_LATENCY_IMPROVEMENT * 100.0
            ))
        } else if improvement > 0.0 {
            EvalResult::warn(test_name, format!(
                "latency improved {:.1}% ({}ms -> {:.0}ms avg), < {:.0}% threshold",
                improvement * 100.0, first as u128, last_avg, MIN_LATENCY_IMPROVEMENT * 100.0
            ))
        } else {
            EvalResult::warn(test_name, format!(
                "latency did not improve: {}ms -> {:.0}ms (hot tier may need more data)",
                first as u128, last_avg
            ))
        }
    } else {
        EvalResult::warn(test_name, "first query too fast to measure improvement")
    }
}

fn test_tiered_metrics(client: &mut McpClient) -> EvalResult {
    let test_name = "tiered_metrics";

    let metrics = client.call_tool("memory.metrics", serde_json::json!({}));

    match metrics {
        Ok(m) => {
            if let Some(tiered) = m.get("tiered") {
                // Verify required fields are present
                let has_cache_hits = tiered.get("cache_hits").is_some();
                let has_cache_misses = tiered.get("cache_misses").is_some();
                let has_cache_hit_rate = tiered.get("cache_hit_rate").is_some();
                let has_hot_searches = tiered.get("hot_searches").is_some();
                let has_promotions = tiered.get("promotions").is_some();

                if has_cache_hits && has_cache_misses && has_cache_hit_rate && has_hot_searches && has_promotions {
                    EvalResult::pass(test_name, "all required tiered metrics present")
                } else {
                    let missing: Vec<&str> = [
                        (!has_cache_hits).then_some("cache_hits"),
                        (!has_cache_misses).then_some("cache_misses"),
                        (!has_cache_hit_rate).then_some("cache_hit_rate"),
                        (!has_hot_searches).then_some("hot_searches"),
                        (!has_promotions).then_some("promotions"),
                    ].into_iter().flatten().collect();
                    EvalResult::fail(test_name, format!("missing tiered metrics: {:?}", missing))
                }
            } else {
                EvalResult::fail(test_name, "tiered section missing from metrics response")
            }
        }
        Err(e) => EvalResult::fail(test_name, format!("metrics call failed: {}", e)),
    }
}
```
