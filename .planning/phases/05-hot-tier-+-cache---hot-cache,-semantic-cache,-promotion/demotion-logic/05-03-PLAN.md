```rust
pub async fn search_with_timing(
    &self,
    tenant_id: &TenantId,
    query_embedding: &[f32],
    k: usize,
    active_project: Option<&str>,
) -> Result<(Vec<TieredSearchResult>, TieredSearchTiming)> {
    let total_start = Instant::now();
    let mut timing = TieredSearchTiming::default();

    // Step 1: Check semantic cache
    if self.config.enable_semantic_cache {
        let cache_start = Instant::now();
        let version = self.get_tenant_version(tenant_id);
        if let Some(cached) = self.semantic_cache.get_similar(
            &tenant_id.to_string(),
            query_embedding,
            version,
        ) {
            timing.cache_check_time = cache_start.elapsed();
            timing.cache_hit = true;
            timing.total_time = total_start.elapsed();

            let results = cached.chunk_ids.into_iter()
                .zip(cached.scores.into_iter())
                .map(|(id, score)| TieredSearchResult {
                    chunk_id: id,
                    score,
                    source: SearchSource::SemanticCache,
                })
                .collect();

            return Ok((results, timing));
        }
        timing.cache_check_time = cache_start.elapsed();
    }

    // Step 2: Search hot tier first
    let mut hot_results = Vec::new();
    if self.config.enable_hot_tier {
        let hot_start = Instant::now();
        let hot_tier = self.get_or_create_hot_tier(tenant_id)?;
        hot_results = hot_tier.search(query_embedding, k)?
            .into_iter()
            .map(|r| TieredSearchResult {
                chunk_id: r.chunk_id,
                score: r.score,
                source: SearchSource::HotTier,
            })
            .collect();
        timing.hot_search_time = hot_start.elapsed();

        // Record access for all hot results
        for result in &hot_results {
            hot_tier.record_access(&result.chunk_id, active_project);
        }
    }

    // Step 3: If hot has enough results, use them
    if hot_results.len() >= k {
        timing.hot_sufficient = true;
        timing.total_time = total_start.elapsed();
        return Ok((hot_results, timing));
    }

    // Step 4: Fall back to warm tier for remaining
    let warm_start = Instant::now();
    let warm_results = self.warm_searcher
        .search(tenant_id, query_embedding, k)
        .await?;
    timing.warm_search_time = warm_start.elapsed();

    // Merge hot and warm, dedupe by chunk_id, take top k
    let merged = merge_results(hot_results, warm_results, k);

    // Record access for warm results (for future promotion)
    let hot_tier = self.get_or_create_hot_tier(tenant_id)?;
    for result in &merged {
        if result.source == SearchSource::WarmTier {
            hot_tier.record_access(&result.chunk_id, active_project);
        }
    }

    // Cache results for similar queries
    if self.config.enable_semantic_cache {
        let cached_result = CachedQueryResult {
            chunk_ids: merged.iter().map(|r| r.chunk_id.clone()).collect(),
            scores: merged.iter().map(|r| r.score).collect(),
            tenant_version: self.get_tenant_version(tenant_id),
            cached_at_ms: now_ms(),
        };
        self.semantic_cache.insert(&tenant_id.to_string(), query_embedding.to_vec(), cached_result);
    }

    timing.total_time = total_start.elapsed();
    Ok((merged, timing))
}
```

Helper function:
```rust
fn merge_results(
    hot: Vec<TieredSearchResult>,
    warm: Vec<SearchResult>,  // From DenseSearcher
    k: usize,
) -> Vec<TieredSearchResult> {
    let mut seen = HashSet::new();
    let mut merged = Vec::with_capacity(k);

    // Hot results first (already sorted by score)
    for r in hot {
        if seen.insert(r.chunk_id.to_string()) {
            merged.push(r);
        }
    }

    // Add warm results not in hot
    for r in warm {
        if merged.len() >= k { break; }
        if seen.insert(r.chunk_id.to_string()) {
            merged.push(TieredSearchResult {
                chunk_id: r.chunk_id,
                score: r.score,
                source: SearchSource::WarmTier,
            });
        }
    }

    merged
}
```

```rust
/// Promote a chunk from warm to hot tier
/// Called when access pattern indicates chunk should be hot
pub fn promote_chunk(
    &self,
    tenant_id: &TenantId,
    chunk_id: &ChunkId,
    embedding: &[f32],
) -> Result<()> {
    let hot_tier = self.get_or_create_hot_tier(tenant_id)?;

    // Check if already in hot
    if hot_tier.contains(chunk_id) {
        return Ok(());
    }

    // Check if hot tier is full
    if hot_tier.len() >= self.config.hot_tier.max_chunks {
        // Evict least valuable chunk
        let now_ms = now_ms();
        if let Some(evict_id) = hot_tier.get_eviction_candidates(now_ms, 1).first() {
            hot_tier.remove(evict_id)?;
            tracing::debug!(chunk_id = %evict_id, "evicted chunk from hot tier");
        }
    }

    hot_tier.insert(chunk_id, embedding)?;
    tracing::debug!(chunk_id = %chunk_id, "promoted chunk to hot tier");

    Ok(())
}

/// Check and promote chunks that meet threshold
/// Called periodically or after search
pub fn check_promotions(
    &self,
    tenant_id: &TenantId,
    active_project: Option<&str>,
    get_embedding: impl Fn(&ChunkId) -> Option<Vec<f32>>,
) -> Result<usize> {
    let hot_tier = self.get_or_create_hot_tier(tenant_id)?;
    let candidates = hot_tier.get_promotion_candidates(active_project);

    let mut promoted = 0;
    for chunk_id in candidates {
        if let Some(embedding) = get_embedding(&chunk_id) {
            self.promote_chunk(tenant_id, &chunk_id, &embedding)?;
            promoted += 1;
        }
    }

    Ok(promoted)
}

/// Evict chunks from hot tier that are no longer valuable
/// Called periodically to clean up cold chunks
pub fn evict_cold_chunks(&self, tenant_id: &TenantId, max_evictions: usize) -> Result<usize> {
    let hot_tier = self.get_or_create_hot_tier(tenant_id)?;
    let now_ms = now_ms();

    let candidates = hot_tier.get_eviction_candidates(now_ms, max_evictions);
    let mut evicted = 0;

    for chunk_id in candidates {
        hot_tier.remove(&chunk_id)?;
        tracing::debug!(chunk_id = %chunk_id, "evicted cold chunk from hot tier");
        evicted += 1;
    }

    Ok(evicted)
}
```

```rust
#[cfg(test)]
mod tests {
    use super::*;
    use crate::embeddings::MockEmbedder;
    use crate::store::DenseSearchConfig;
    use std::sync::Arc;

    fn make_test_tiered_searcher() -> TieredSearcher {
        let embedder = Arc::new(MockEmbedder::new());
        let dense_config = DenseSearchConfig {
            persist: false,
            ..Default::default()
        };
        let warm = Arc::new(DenseSearcher::with_embedder(embedder, dense_config));

        TieredSearcher::new(warm, TieredSearchConfig::default())
    }

    fn normalize(v: &mut [f32]) {
        let norm: f32 = v.iter().map(|x| x * x).sum::<f32>().sqrt();
        if norm > 0.0 {
            for x in v.iter_mut() {
                *x /= norm;
            }
        }
    }

    #[tokio::test]
    async fn test_tiered_search_hot_then_warm() {
        let searcher = make_test_tiered_searcher();
        let tenant = TenantId::new("test").unwrap();

        // Add chunk to hot tier
        let chunk_id = ChunkId::new();
        let mut embedding = vec![1.0, 0.0, 0.0, 0.0];
        normalize(&mut embedding);

        searcher.promote_chunk(&tenant, &chunk_id, &embedding).unwrap();

        // Search should find it in hot tier
        let (results, timing) = searcher
            .search_with_timing(&tenant, &embedding, 10, None)
            .await
            .unwrap();

        assert!(!results.is_empty());
        assert_eq!(results[0].source, SearchSource::HotTier);
        assert!(timing.hot_sufficient || timing.warm_search_time > Duration::ZERO);
    }

    #[test]
    fn test_promotion_eviction_cycle() {
        let searcher = make_test_tiered_searcher();
        let tenant = TenantId::new("test").unwrap();

        // Promote chunks up to limit
        for i in 0..10 {
            let chunk_id = ChunkId::new();
            let mut embedding = vec![i as f32, 0.0, 0.0, 0.0];
            normalize(&mut embedding);
            searcher.promote_chunk(&tenant, &chunk_id, &embedding).unwrap();
        }

        let stats = searcher.get_stats(&tenant);
        assert!(stats.hot_tier_size <= searcher.config.hot_tier.max_chunks);
    }
}
```
