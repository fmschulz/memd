```rust
    pub source: FusionSource,

    pub source_score: f32,  // Original score from source
}

/// Configuration for RRF
#[derive(Debug, Clone)]
pub struct RrfConfig {
    /// RRF constant k (default 60, standard value from literature)
    pub k: f32,
    /// Weight for dense source (default 1.0)
    pub dense_weight: f32,
    /// Weight for sparse source (default 1.0)
    pub sparse_weight: f32,
}

/// Fused result
pub struct FusedResult {
    pub chunk_id: ChunkId,
    pub rrf_score: f32,
    pub dense_rank: Option<usize>,
    pub sparse_rank: Option<usize>,
}
```

Implement RrfFusion:
```rust
pub struct RrfFusion {
    config: RrfConfig,
}

impl RrfFusion {
    pub fn new(config: RrfConfig) -> Self;

    /// Fuse candidates from multiple sources using RRF
    /// RRF score = sum over sources: weight / (k + rank)
    pub fn fuse(&self, candidates: Vec<FusionCandidate>) -> Vec<FusedResult>;
}
```

RRF algorithm:
1. Group candidates by chunk_id
2. For each chunk, compute RRF score:
   - If present in dense at rank r: add dense_weight / (k + r)
   - If present in sparse at rank r: add sparse_weight / (k + r)
3. Sort by RRF score descending
4. Return FusedResult with chunk_id, rrf_score, and source ranks

```rust
/// Configuration for reranker
#[derive(Debug, Clone)]
pub struct RerankerConfig {
    /// Weight for RRF score (default 1.0)
    pub rrf_weight: f32,
    /// Weight for recency bonus (default 0.1)
    pub recency_weight: f32,
    /// Decay half-life in days for recency (default 7.0)
    pub recency_half_life_days: f32,
    /// Weight for project match bonus (default 0.2)
    pub project_weight: f32,
    /// Weight for type match bonus (default 0.05)
    pub type_weight: f32,
}

/// Input chunk with metadata for reranking
pub struct ChunkWithMeta {
    pub chunk_id: ChunkId,
    pub rrf_score: f32,
    pub timestamp_created: i64,
    pub project_id: Option<String>,
    pub chunk_type: ChunkType,
}

/// Context for reranking (query-time info)
pub struct RerankerContext {
    pub current_project: Option<String>,
    pub preferred_types: Vec<ChunkType>,
    pub now_ms: i64,  // Current timestamp for recency calc
}

/// Result with final score
pub struct RankedResult {
    pub chunk_id: ChunkId,
    pub final_score: f32,
    pub rrf_score: f32,
    pub recency_bonus: f32,
    pub project_bonus: f32,
    pub type_bonus: f32,
}
```

**FeatureReranker:**
```rust
pub struct FeatureReranker {
    config: RerankerConfig,
}

impl FeatureReranker {
    pub fn new(config: RerankerConfig) -> Self;

    /// Rerank chunks using feature-based scoring
    pub fn rerank(
        &self,
        chunks: Vec<ChunkWithMeta>,
        context: &RerankerContext
    ) -> Vec<RankedResult>;
}
```

**Feature calculations:**

1. **Recency bonus:**
   - age_days = (now_ms - timestamp_created) / (1000 * 60 * 60 * 24)
   - recency = exp(-age_days * ln(2) / half_life)
   - Range: 1.0 (just created) to 0.0 (very old)

2. **Project match bonus:**
   - 1.0 if chunk.project_id matches context.current_project
   - 0.0 otherwise

3. **Type match bonus:**
   - 1.0 if chunk.chunk_type in context.preferred_types
   - 0.0 otherwise

4. **Final score:**
   - final = rrf_weight * rrf_score
           + recency_weight * recency_bonus
           + project_weight * project_bonus
           + type_weight * type_bonus

5. Sort by final_score descending
