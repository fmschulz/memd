pub enum QueryIntent {
    // Semantic search (default)
    SemanticSearch,

    // Structural code queries
    SymbolDefinition(String),   // Find where X is defined
    SymbolReferences(String),   // Find usages of X
    SymbolCallers(String),      // Find who calls X
    ModuleImports(String),      // Find who imports X
    FileSymbols(String),        // All symbols in file X

    // Debug/trace queries
    ToolCalls(Option<String>),  // Tool calls, optionally filtered
    ErrorSearch(Option<String>), // Errors, optionally by signature

    // Document queries
    DocQa,                      // Documentation question
    DecisionWhy(String),        // Why was X decided
    PlanNext,                   // What's the next step
}
```

  - `(?i)what calls (\w+)`
  - `(?i)(\w+) is called by`

  Reference patterns:
  - `(?i)references to (\w+)`
  - `(?i)usages of (\w+)`
  - `(?i)where is (\w+) used`
  - `(?i)find all (?:uses|usages|references) (?:of )?(\w+)`

  Import patterns:
  - `(?i)who imports (\w+)`
  - `(?i)files (?:that )?import (\w+)`
  - `(?i)(\w+) importers`

  Error patterns:
  - `(?i)errors? (?:in|from|about) (\w+)`
  - `(?i)stack traces? (?:for|with) (\w+)`
  - `(?i)exceptions? (?:in|from) (\w+)`
  - `(?i)what went wrong`  -- general error search

  Tool patterns:
  - `(?i)(?:tool )?calls to (\w+)`
  - `(?i)when did (?:we|I) (?:call|use) (\w+)`
  - `(?i)history of (\w+) calls`
  - `(?i)recent tool calls`

- classify(&self, query: &str) -> QueryIntent
  1. Check explicit prefixes first (override pattern detection):
     - "def:" -> SymbolDefinition
     - "callers:" -> SymbolCallers
     - "refs:" -> SymbolReferences
     - "imports:" -> ModuleImports
     - "errors:" -> ErrorSearch
     - "tools:" -> ToolCalls

  2. Try each pattern category in priority order:
     - Definition patterns
     - Caller patterns
     - Reference patterns
     - Import patterns
     - Error patterns
     - Tool patterns

  3. If no pattern matches, return SemanticSearch

- extract_target(pattern: &Regex, query: &str, group: usize) -> Option(String)
  - Run regex, extract capture group
  - Normalize target (trim, etc.)

    /// Returns whether this intent should blend semantic context with structural results
    /// Per CONTEXT.md: "structural first, semantic expands"
    pub fn should_blend_semantic(&self) -> bool {

            QueryIntent::SymbolDefinition(_) |
            QueryIntent::SymbolReferences(_) |
            QueryIntent::SymbolCallers(_) |
            QueryIntent::ModuleImports(_) |
            QueryIntent::FileSymbols(_)

}

pub enum SearchWithRoutingResult {
    /// Pure hybrid/semantic search results
    Hybrid(Vec<RankedResult>),

    /// Structural results only (no semantic context)
    Structural(StructuralSearchResult),

    /// Structural primary + semantic context (STRUCT-14 blending)
    Blended(BlendedSearchResult),

    /// Trace/debug results
    Trace(TraceSearchResult),
}

pub struct StructuralSearchResult {
    intent: QueryIntent,
    results: Vec<SymbolLocation>,  // or CallerInfo, ImportInfo
    fell_back_to_semantic: bool,
}

pub struct BlendedSearchResult {
    intent: QueryIntent,
    /// Primary results from structural search
    structural_results: Vec<SymbolLocation>,
    /// Supplementary context from semantic search
    semantic_context: Vec<RankedResult>,
    /// How results were weighted
    blend_strategy: BlendStrategy,
}

pub enum BlendStrategy {
    /// Structural results first (100%), semantic for context only
    StructuralPrimary,
    /// Weighted blend (e.g., 70% structural, 30% semantic by relevance)
    Weighted { structural_weight: f32 },
}

pub struct TraceSearchResult {
    intent: QueryIntent,
    tool_calls: Vec<ToolCallResult>,
    errors: Vec<ErrorResult>,
}
```

**Blending logic (STRUCT-14 implementation):**
```rust
fn blend_results(
    structural: Vec<SymbolLocation>,
    semantic: Vec<RankedResult>,
    strategy: BlendStrategy,
) -> BlendedSearchResult {
    match strategy {
        BlendStrategy::StructuralPrimary => {
            // Return all structural results first
            // Append semantic results as "Related Context" section
            BlendedSearchResult {
                structural_results: structural,
                semantic_context: semantic,
                blend_strategy: strategy,
                ..
            }
        }
        BlendStrategy::Weighted { structural_weight } => {
            // For future: interleave based on weights
            // Current implementation: structural primary
            BlendedSearchResult {
                structural_results: structural,
                semantic_context: semantic,
                blend_strategy: strategy,
                ..
            }
        }
    }
}
```
